# Write a Python program to test whether a number is within 100 of 1000 or 2000.

print("This program will test whether a number is within 100 of 1000 or 2000")

num = int(input("Enter a number:"))

if (num <= 1100 and num >= 900) or (num <= 2100 and num >= 1900):
    print("True, This number is is within 100 of 1000 or 2000")
else:
    print("False, This number is NOT within 100 of 1000 or 2000")
